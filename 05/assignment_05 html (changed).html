solve a linear system of equation $A z = b$¶

$$ A = \begin{bmatrix} x_1^0 & x_1^1 & \cdots & x_1^{p-1}\\ x_2^0 & x_2^1 & \cdots & x_2^{p-1}\\ \vdots & \vdots & \vdots & \vdots\\ x_n^0 & x_n^1 & \cdots & x_n^{p-1} \end{bmatrix}, \quad z = \begin{bmatrix} \theta_0\\ \theta_1\\ \vdots\\ \theta_{p-1} \end{bmatrix}, \quad b = \begin{bmatrix} y_1\\ y_2\\ \vdots\\ y_n \end{bmatrix} $$

construct matrix A for the polynomial regression with power $p-1$¶

  • useful functions : np.power
In [ ]:
def construct_matrix_A(x, p):
    
    n = len(x)
    A = np.zeros([n, p])
    
    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    
    # for i in range(n):
    #     for j in range(p):
    #         A[i][j]=np.power(x[i],j)

    #for문 안쓰고 만드는 법 찾기
    square = np.arange(p)    #p[0,1,2,3,4,5,6,7,8,...,p-1]
    for i in range (n):
        A[i,:] = np.power(x[i],square)
    # print(A)
    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return A

solve the linear system of equation $A z = b$¶

  • without regularization : $\min \frac{1}{2 n}\| Az - b \|^2, \quad z = \left( A^T A \right)^{-1} A^T b$
  • useful functions : np.matmul, np.linalg.inv, np.sum
In [ ]:
def solve_regression(x, y, p):

    z       = np.zeros([p, 1])
    loss    = 0

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    # 맞는지 확인하기
    ATA = np.matmul(construct_matrix_A(x,p).T, construct_matrix_A(x,p))
    inv_ATA = np.linalg.inv(ATA)
    z_temp = np.matmul(inv_ATA, construct_matrix_A(x,p).T)
    z=np.matmul(z_temp,construct_vector_b(y))
    L2 = np.matmul(construct_matrix_A(x,p), z)-construct_vector_b(y)
    # L2norm = np.linalg.norm(L2)
    # print(L2norm)
    # L2norm_square = np.power(L2norm, 2)
    # print(L2norm_square)
    # loss = (np.sum(L2norm_square))/(2*len(x))
    # print(np.sum(L2norm_square))
    loss = np.sum(np.power(L2,2))/(2*len(x))

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return z, loss
  • with regularization : $\min \frac{1}{2 n} \| A z - b \|^2 + \frac{\alpha}{2} \| z \|^2, \quad z = \left(A^{T} A + n \alpha I \right)^{-1} A^T b$ where $I$ denotes identity matrix
  • useful functions : np.matmul, np.linalg.inv, np.sum

def solve_regression_with_regularization(x, y, p, alpha):

    z       = np.zeros([p, 1])
    loss    = 0

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    n=len(x)
    naI = n*alpha*np.eye(p)
    ATA = np.matmul(construct_matrix_A(x,p).T, construct_matrix_A(x,p))
    ATA_temp = ATA+naI
    inv_ATA = np.linalg.inv(ATA_temp)
    z_temp = np.matmul(inv_ATA, construct_matrix_A(x,p).T)
    z=np.matmul(z_temp,construct_vector_b(y))
    
    #loss
    L2 = np.matmul(construct_matrix_A(x,p), z)-construct_vector_b(y)
    # L2norm = np.linalg.norm(L2)
    # L2norm_square = np.power(L2norm, 2)
    # lossL2 = np.sum(L2norm_square)/(2*len(x))
    lossL2 = np.sum(np.power(L2,2))/(2*len(x))
    # znorm = np.linalg.norm(z)
    # znorm_square = np.power(znorm,2)
    # loss_znorm = np.sum(znorm_square)*alpha/2
    loss_znorm = np.sum(np.power(z,2))*alpha/2
    loss = lossL2+loss_znorm
    
    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return z, loss

def approximate(x, y, p):

    n       = len(y)
    y_hat   = np.zeros([n, 1])
    loss    = 0

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #y_hat = Az*
    
    reg_z, reg_loss = solve_regression(x,y,p) 
    y_hat = np.matmul(construct_matrix_A(x,p), reg_z)
    loss = reg_loss
    
    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return y_hat, loss

def approximate_with_regularization(x, y, p, alpha):
    
    n       = len(y)
    y_hat   = np.zeros([n, 1])
    loss    = 0

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    reg_reg_z, reg_reg_loss = solve_regression_with_regularization(x,y,p,alpha)
    y_hat = np.matmul(construct_matrix_A(x,p), reg_reg_z)
    loss = reg_reg_loss
    
    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return y_hat, loss
